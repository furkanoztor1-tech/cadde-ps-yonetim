`; if (d.tip === "Gider") gLog += item; else if (d.tip === "Borç") bLog += item; else log += item; }); document.getElementById('vT').innerText = kasa + "₺"; document.getElementById('islem-log').innerHTML = log || "
İşlem yok.
"; document.getElementById('gider-log').innerHTML = gLog || "

"; document.getElementById('borc-log').innerHTML = bLog || "

"; }, err => { console.error("Veri çekme hatası:", err); }); unsubs.push(u1); } async function gunuKapat() { const sName = document.getElementById('i-sube').value; if (!confirm(sName + " Z Raporu alınıp kasa sıfırlansın mı?")) return; try { const snap = await db.collection("islemler") .where("sube", "==", sName) .where("durum", "==", "aktif") .where("arsiv", "==", false) .get(); if (snap.empty) { alert("Kapatılacak kayıt bulunamadı!"); return; } let n = 0, k = 0, h = 0, g = 0, b = 0, ids_to_archive = []; snap.forEach(doc => { const d = doc.data(); const m = Number(d.miktar) || 0; if (d.tip === "Nakit") n += m; else if (d.tip === "Kart") k += m; else if (d.tip === "Havale") h += m; else if (d.tip === "Gider") g += m; else if (d.tip === "Borç") b += m; // Borçlar hariç her şeyi arşive işaretle if (d.tip !== "Borç") { ids_to_archive.push(doc.id); } }); // Z Raporunu Kaydet await db.collection("z_raporlari").add({ tarih: new Date().toLocaleString('tr-TR'), sube: sName, kapatan: currentUserName, nakit: n, kart: k, havale: h, gider: g, borc: b, net: (n + k + h) - g, ts: new Date() }); // Toplu Arşivleme const batch = db.batch(); ids_to_archive.forEach(id => { batch.update(db.collection("islemler").doc(id), { arsiv: true }); }); await batch.commit(); alert("Z Raporu Başarıyla Alındı!"); location.reload(); } catch (error) { console.error("Hata:", error); alert("İşlem sırasında bir hata oluştu!"); } }
